CREATE VIEW 学部名付き学生 AS//このSELECT結果を『学部名付き学生』という名前で保存する
//学生テーブル（S）から必要な情報だけ取る
SELECT S.学籍番号,S.名前,S.生年月日,S.血液型
//学部の名前を“学部名”というカラム名で表示
S.学部ID,B.名前　AS 学部名 
//ベースは学生テーブル AS S → 学生をSって呼ぶ
FROM 学生 AS S
//別のテーブル（B）とくっつける
JOIN 学籍番号 AS B
//学部IDが一致するデータ同士を結合
ON S.学部ID = B.ID
